69. 대여 횟수가 많은 자동차들의 월별 대여 횟수 구하기

https://school.programmers.co.kr/learn/courses/30/lessons/151139

SELECT
    START_DATE MONTH,
    CAR_ID,
    COUNT(1) RECORDS
FROM (
    SELECT
        HISTORY_ID,
        CAR_ID,
        DATE_FORMAT(START_DATE, '%m') START_DATE
    FROM CAR_RENTAL_COMPANY_RENTAL_HISTORY
    ) A
WHERE CAR_ID IN (
    SELECT
        CAR_ID
    FROM CAR_RENTAL_COMPANY_RENTAL_HISTORY
    WHERE DATE_FORMAT(START_DATE, '%m') BETWEEN 08 AND 10
    GROUP BY CAR_ID
    HAVING COUNT(1)>=5) AND START_DATE BETWEEN 08 AND 10
GROUP BY 1, 2
ORDER BY 1, 2 DESC
